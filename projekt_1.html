for (col in string_columns)
{
  col_counts <- data %>%
    count(get(col), name = "Count") %>%
    arrange(desc(Count))
  col_counts$var <- col
  counts <- rbind(counts, col_counts)
  
  top_10_counts <- col_counts %>%
    slice(1:10)
